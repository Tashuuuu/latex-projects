\noindent \textbf{Q1. Using pstricks package make lines, a vector, triangle, circle, square and rectangle.}
\begin{enumerate}
\item Line:\\
\begin{pspicture}(4,4)
\psline(0,0)(10,4)
\end{pspicture}
	
        \item Dotted line:\\
        \begin{pspicture}(3,5)
        	\psline[linestyle=dotted,linewidth=0.07](0,0)(6,4)
        \end{pspicture}	
    
		\item Vector:\\ 
		\begin{pspicture}(4,4)
			\psline{->}(0,0)(2.5,3.99)
		\end{pspicture}	
		
		\item Isosceles triangle:\\
		\begin{pspicture}(13,13)
			\psline(0,0)(12,10)(9,2)(0,0)
		\end{pspicture}
\newpage
		\item Circle:\\
		\begin{pspicture}(5,5)
			\pscircle(1.888889,2.8){1.99999}
		\end{pspicture}
	
	    \item Square:\\
	    \begin{pspicture}(4.4,4.4)
	    	\psline(0.2,0.2)(0.2,4)(4,4)(4,0.2)(0.2,0.2)
	    \end{pspicture}
	   
		\item Rectangle:\\
		\begin{pspicture}(8,8)
			\psline(0.25,7.5)(12,7.5)(12,2)(0.25,2)(0.25,7.5)
		\end{pspicture}
	\end{enumerate}
